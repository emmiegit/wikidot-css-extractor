  • /* Code de message basique */
.text-container .text {
  border-radius: 18px;
  text-align: left;
  display:inline-block;
  padding:8px 12px;
  font-weight:500;
  margin:1.5px 10px;
  max-width: 300px;
}
 
/* Bloc des messages "reçus" */
.recv {
  text-align: left;
  margin-right:20px;
}
 
/* Bloc des messages "envoyés" */
.sent {
  text-align: right;
  margin-left:20px;
}
.sent .text {
  border-bottom-right-radius: 3px;
  background-color: #267AE9;
  color: white;
}
.recv .text {
  border-bottom-left-radius: 3px;
  background-color: #dadada;
  color: black;
}
 
/* Truc stylé pour les coins des messages quand il y en a un autre au-dessus */
.sent .text:not(:first-child) {
  border-top-right-radius: 4px;
}
.recv .text:not(:first-child) {
  border-top-left-radius: 4px;
}
 
.text-container p {
  margin:0 0 15px;
}
/* Là où les messages s'affichent */
/* Style du header */
.text-container-wrap h1, .text-container-wrap h2, .text-container-wrap h3, .text-container-wrap h4, .text-container-wrap h5, .text-container-wrap h6 {
  color: black;
  font-weight: bold;
  margin:0 auto 10px;
  position:relative;
  top:-8px;
}
 
/* Boite externe */
.text-container-wrap {
  text-align:center;
  margin:auto;
  display:block;
  margin-top: 25px;
  width:400px;
  padding-top: 25px;

  background-color:#fff;
  overflow:none;
  box-sizing:border-box;
  pointer-events:none!important;
  position:relative;
}
 
/* Faire croire qu'on ne peut pas sélectionner le texte (même si on peut), pour la crédibilité */
.text-container-wrap *::selection {
  background-color:transparent!important;
  color:inherit!important;
}
 
/* Boite interne */
