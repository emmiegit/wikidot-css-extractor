}

    display: block;

}

    padding-bottom: 5em;
    text-align: justify; /* alignement */
    font-family: 'Cormorant', serif; /* police */
    font-size:17px; /* taille */
}

.1er_paragrapheANF p:first-letter { /* première lettre en majuscule pour le premier paragraphe du texte (optionnel) */
  font-size: 600%; /* taille de la première lettre */
  float: left;
}

.texte_gauche_pageANF { /* position du texte quand il est sur la page gauche */
    padding-left:3em;
    padding-right:5em;
    transform: rotate(-0.1deg);

}

.texte_droite_pageANF { /* position du texte quand il est sur la page de droite */

    padding-right:4.5em;
    transform: rotate(0.1deg);
    text-shadow: 0.25px 0px 0.25px #e1e4cf;
}


/* Bandeau ANF */

.bandeauANF { /* Bandeau supérieure de la page */
    position: relative;
    text-align: center;
}

.bandeauANF .gauche { left: 0; }    /* Éléments du bandeau à gauche */
.bandeauANF .droite { right: 0; }  /* Éléments du bandeau à droite */

.bandeauANF .gauche, .bandeauANF .droite {
  position: absolute;
  top: 50%;
  transform: translate(0, -50%);
}

.bandeauANF .centre {
  font-variant: small-caps;
}
